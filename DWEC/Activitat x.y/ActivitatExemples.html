Loop For

Mostrar amb un For el cotxes recorrent un array

			<p id="cotxes"></p>

			<script>
			cars = ["BMW", "Volvo", "Saab", "Ford"];
			text = "";
			var i;
			for (i = 0; i < cars.length; i++) {
			text += cars[i] + "<br>";
			}

			document.getElementById("cotxes").innerHTML = text;
			</script>
